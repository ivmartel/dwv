/**
 * Lookup tables for image colour display.
 */

const lut_range_max = 256;

/**
 * Build a LUT of size lut_range_max.
 *
 * @param {Function} func The i to lut function.
 * @returns {number[]} The LUT.
 */
function buildLut(func) {
  const lut = [];
  for (let i = 0; i < lut_range_max; ++i) {
    lut.push(func(i));
  }
  return lut;
}

/**
 * Ramp to lut_range_max minus one on the first third values.
 *
 * @param {number} i The input index.
 * @returns {number} The lut value.
 */
function toMaxFirstThird(i) {
  const val = i * 3;
  if (val > lut_range_max - 1) {
    return lut_range_max - 1;
  }
  return val;
}

/**
 * Ramp to lut_range_max minus one on the second third values,
 *   otherwise return 0 for the first third and
 *   lut_range_max minus one for the last third.
 *
 * @param {number} i The input index.
 * @returns {number} The lut value.
 */
function toMaxSecondThird(i) {
  const third = lut_range_max / 3;
  let val = 0;
  if (i >= third) {
    val = (i - third) * 3;
    if (val > lut_range_max - 1) {
      return lut_range_max - 1;
    }
  }
  return val;
}

/**
 * Ramp to lut_range_max minus one on the last third values,
 *   otherwise return 0.
 *
 * @param {number} i The input index.
 * @returns {number} The lut value.
 */
function toMaxThirdThird(i) {
  const third = lut_range_max / 3;
  let val = 0;
  if (i >= 2 * third) {
    val = (i - 2 * third) * 3;
    if (val > lut_range_max - 1) {
      return lut_range_max - 1;
    }
  }
  return val;
}

/**
 * Identity, returns i.
 *
 * @param {number} i The input index.
 * @returns {number} The lut value.
 */
function id(i) {
  return i;
}

/**
 * Returns lut_range_max minus one minus i.
 *
 * @param {number} i The input index.
 * @returns {number} The lut value.
 */
function invId(i) {
  return (lut_range_max - 1) - i;
}
